### Uso

```js
import { TileGroup } from 'nr1'
```

### Exemplos

#### Grupo básico de blocos

```js
<TileGroup>
  <Tile onClick={console.log}>Tile 1</Tile>
  <Tile onClick={console.log}>Tile 2</Tile>
  <Tile onClick={console.log}>Tile 3</Tile>
</TileGroup>
```

#### Grupo de ladrilhos com tipo de lacuna pequena

```js
<TileGroup gapType={TileGroup.GAP_TYPE.SMALL}>
  <Tile onClick={console.log}>Tile 1</Tile>
  <Tile onClick={console.log}>Tile 2</Tile>
  <Tile onClick={console.log}>Tile 3</Tile>
</TileGroup>
```

#### Grupo de blocos com 3 colunas

```js
<TileGroup tileWidth="4fr">
  {Array.from({ length: 12 }).map((_, index) => (
    <Tile onClick={console.log}>Tile {index + 1}</Tile>
  ))}
</TileGroup>
```

#### Grupo de blocos com largura de bloco personalizada

```js
<TileGroup tileWidth={120}>
  {Array.from({ length: 12 }).map((_, index) => (
    <Tile onClick={console.log}>Tile {index + 1}</Tile>
  ))}
</TileGroup>
```

#### Blocos selecionáveis

```js
<TileGroup selectionType={TileGroup.SELECTION_TYPE.SINGLE} tileWidth={120}>
  {Array.from({ length: 12 }).map((_, index) => (
    <Tile onClick={console.log}>Tile {index + 1}</Tile>
  ))}
</TileGroup>
```

#### Blocos selecionáveis com bloco padrão marcado

```js
<TileGroup
  defaultValue="tile-0"
  selectionType={TileGroup.SELECTION_TYPE.SINGLE}
  tileWidth={120}
>
  {Array.from({ length: 12 }).map((_, index) => (
    <Tile value={`tile-${index}`} onClick={console.log}>
      Tile {index + 1}
    </Tile>
  ))}
</TileGroup>
```

#### Blocos selecionáveis controlados

```js
class MyComponent extends React.PureComponent {
  constructor() {
    super(...arguments);
    this.state = {
      checkedTile: 'tile-0',
    };
    this._onChange = this._onChange.bind(this);
  }
  _onChange(evt, tileValue) {
    this.setState({ checkedTile: tileValue });
  }
  render() {
    const { checkedTile } = this.state;
    return (
      <TileGroup
        value={checkedTile}
        onChange={this._onChange}
        tileWidth={120}
        selectionType={TileGroup.SELECTION_TYPE.SINGLE}
      >
        {Array.from({ length: 12 }).map((_, index) => (
          <Tile
            sizeType={Tile.SIZE_TYPE.SMALL}
            value={`tile-${index}`}
            key={index}
            onClick={console.log}
          >
            Tile {index + 1}
          </Tile>
        ))}
      </TileGroup>
    );
  }
}
```

#### Blocos de seleção múltipla

```js
<TileGroup selectionType={TileGroup.SELECTION_TYPE.MULTIPLE} tileWidth={120}>
  {Array.from({ length: 12 }).map((_, index) => (
    <Tile onClick={console.log}>Tile {index + 1}</Tile>
  ))}
</TileGroup>
```

#### Blocos de seleção múltipla controlados

```js
class MyComponent extends React.PureComponent {
  constructor() {
    super(...arguments);
    this.state = {
      checkedTiles: [],
    };
    this._onChange = this._onChange.bind(this);
  }
  _onChange(evt, tileValue, checked) {
    this.setState((state) => {
      return {
        checkedTiles: checked
          ? [...state.checkedTiles, tileValue]
          : state.checkedTiles.filter((value) => value !== tileValue),
      };
    });
  }
  render() {
    const { checkedTiles } = this.state;
    return (
      <TileGroup
        value={checkedTiles}
        onChange={this._onChange}
        tileWidth={120}
        selectionType={TileGroup.SELECTION_TYPE.MULTIPLE}
      >
        {Array.from({ length: 12 }).map((_, index) => (
          <Tile
            sizeType={Tile.SIZE_TYPE.SMALL}
            value={`tile-${index}`}
            key={index}
            onClick={console.log}
          >
            Tile {index + 1}
          </Tile>
        ))}
      </TileGroup>
    );
  }
}
```
