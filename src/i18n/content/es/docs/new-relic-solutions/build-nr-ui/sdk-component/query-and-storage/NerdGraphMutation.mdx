Un componente de mutación genérico de NerdGraph que le permite mutar cualquier cosa de NerdGraph.

### Uso

```js
import { NerdGraphMutation } from 'nr1'
```

### Ejemplos

#### hacer una mutación

```js
NerdGraphMutation.mutate({
  mutation: ngql`
    mutation($guid: EntityGuid!) {
      taggingAddTagsToEntity(
        guid: $guid
        tags: { key: "team", values: ["ui"] }
      ) {
        errors {
          message
        }
      }
    }
  `,
  variables: {
    guid: 'XXXXXXXXXXX',
  },
});
```

#### Hacer consulta de mutación y recuperación

```js
function render() {
  const mutation = ngql`
    mutation($guid: EntityGuid!) {
      taggingAddTagsToEntity(guid: $guid, tags: $tags) {
        errors {
          message
        }
      }
    }
  `;
  const variables = {
    guid: 'XXXXXXXXXXX',
    tags: { key: 'team', values: ['ui'] },
  };


  // NOTE: Sometimes mutations take awhile so doing a refetch immediatly after a mutate
  // doesn't show any change.
  return (
    <NerdGraphQuery query={query} variables={variables}>
      {({ data, refetch }) => (
        <>
          <RenderData data={data} />
          <Button
            onClick={() =>
              NerdGraphMutation.mutate({
                mutation,
                variables,
              }).then(refetch)
            }
          >
            Mutate
          </Button>
        </>
      )}
    </NerdGraphQuery>
  );
}
```
